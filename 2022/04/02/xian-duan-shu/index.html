leetcode 307. 区域和检索 - 数组可修改

class NumArray {
    int data[];
    int tree[];
    int n;
    public NumArray(int[] nums) {
        n = nums.length;
        data = new int[n];
        tree = new int[4*n];
        for(int i = 0; i < n; i++)
            data[i] = nums[i];
        build(0, 0, n-1);
    }

    void build(int p, int l, int r) {
        if(l == r) {
            tree[p] = data[l];
            return;
        }
        int mid = l + (r-l)/2;
        build(ls(p), l, mid);
        build(rs(p), mid+1, r);
        tree[p] = tree[ls(p)] + tree[rs(p)];
    }
    int query(int p, int l, int r, int s, int e) {
        if(e < l || s > r)
            return 0;
        else if((l==r) || (s <= l && e >= r))
            return tree[p];
        int mid = l + (r-l)/2;
        return query(ls(p), l, mid, s, e) + query(rs(p), mid+1, r, s, e);
    }
    void update(int p, int l, int r, int idx, int val) {
        if(l == r) { //data[idx] = val
            data[idx] = val;
            tree[p] = val;
            return;
        }
        int mid = l + (r-l)/2;
        if(idx >= l && idx <= mid)
            update(ls(p), l, mid, idx, val);
        else
            update(rs(p), mid+1, r, idx, val);
        tree[p] = tree[ls(p)] + tree[rs(p)];
    }
    int ls(int x) { //左
        return 2*x+1;
    }

    int rs(int x) { //右
        return 2*x+2;
    }

    public void update(int index, int val) {
        update(0,0,n-1,index,val);
    }

    public int sumRange(int left, int right) {
        return query(0,0, n-1, left, right);
    }
}
