// 123 -> [3, 2, 1]
static int[] num2arr(String num) {
    int n = num.length(), arr[] = new int[n];
    for(int i = n-1; i >= 0; --i) {
        arr[n-i-1] = num.charAt(i)-'0'; 
    }
    return arr;
}

减法

// A >= B, A >= 0, B >= 0
static List<Integer> sub(int a[], int b[]) {
    List<Integer> ans = new ArrayList<>();
    int t = 0;
    for(int i = 0; i < a.length; i++) {
        t = a[i] - t;
        if(i < b.length) t -= b[i];
        ans.add((t+10) % 10);
        if(t < 0)
            t = 1;
        else
            t = 0;
    }
    while(ans.size() > 1 && ans.get(ans.size()-1) == 0) ans.remove(ans.size()-1);
    return ans;
}

static List<Integer> mul(int a[], int b) {
        List<Integer> ans = new ArrayList<>();
        int t = 0;
        for(int i = 0; i < a.length || t != 0; i++) {
            if(i < a.length) t += a[i] * b;
            ans.add(t % 10);
            t /= 10;
        }
        while(ans.size() > 1 && ans.get(ans.size()-1) == 0) ans.remove(ans.size()-1);
        return ans;

高精度除以低精度

// A >= 0, b > 0
import java.util.*;
public class Main {
    static int t = 0; //  余数  
    static List<Integer> div(String str, int b) {
        int a[] = new int[str.length()];
        for(int i = 0; i < str.length(); i++)
            a[i] = str.charAt(i)-'0';
        List<Integer> ans = new ArrayList<>();
        for(int i = 0; i < a.length; i++) {
            t = t * 10 + a[i];
            ans.add(t / b);
            t %= b;
        }
        Collections.reverse(ans);
        // 去除末尾的0
        while(ans.size() > 1 && ans.get(ans.size()-1) == 0) ans.remove(ans.size()-1);
        return ans;
    }
}
