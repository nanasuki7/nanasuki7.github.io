28. 实现 strStr()

字符串查找，kmp板子题

public int strStr(String haystack, String needle) {
    if(needle.length() == 0)
        return 0;
    char t[] = haystack.toCharArray(), p[] = needle.toCharArray();
    int n = p.length, m = t.length, next[] = new int[n];
    next[0] = -1;
    int i = 0, j = -1;
    while(i < n-1) {
        if(j == -1 || p[i] == p[j]) {
            i++;
            j++;
            next[i] = j;
        }
        else
            j = next[j];
    }
    i = 0; j = 0;
    while(i < m && j < n) {
        if(j == -1 || t[i] == p[j]) {
            i++;
            j++;
        }
        else
            j = next[j];
    }
    if(j == n)
        return i-j;
    return -1;
}
